## 1. Project Overview

ระบบจัดการเครื่องราชอิสริยาภรณ์ (Royal Decorations Management System) สำหรับกระทรวงยุติธรรม ประกอบด้วย 3 สาย + ระบบสนับสนุน:

| Module | ประเภท | กลุ่มเป้าหมาย |
|--------|--------|---------------|
| ช้างเผือก-มงกุฎไทย | เครื่องราชฯ สายหลัก (12 ชั้น) | ข้าราชการ |
| ดิเรกคุณาภรณ์ | เครื่องราชฯ สายพิเศษ (7 ชั้น) | อาสาสมัคร/บุคคลภายนอก |

| File/User/Settings | ระบบสนับสนุน | เจ้าหน้าที่ HR |

```sql
CREATE TABLE direk_persons (

    bankruptcy_check_status VARCHAR(50),
    bankruptcy_check_date   DATE,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- ชั้นตราดิเรกคุณาภรณ์ (7 ชั้น)
CREATE TABLE direk_levels (
    level_id        INT PRIMARY KEY,
    level_name      VARCHAR(100) NOT NULL,
    abbreviation    VARCHAR(20) NOT NULL,
    level_category  VARCHAR(50) NOT NULL,
    min_donation    DECIMAL(15,2),
    form_type       VARCHAR(10),
    sort_order      INT NOT NULL
);

-- ประวัติได้รับพระราชทาน
CREATE TABLE direk_award_history (
    award_id        BIGINT PRIMARY KEY AUTO_INCREMENT,
    person_id       BIGINT NOT NULL,
    level_id        INT NOT NULL,
    award_year      INT,
    gazette_date    DATE,
    gazette_ref     VARCHAR(200),
    request_id      BIGINT,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (person_id) REFERENCES direk_persons(person_id),
    FOREIGN KEY (level_id) REFERENCES direk_levels(level_id)
);

-- คำขอเสนอพระราชทาน
CREATE TABLE direk_requests (
    request_id      BIGINT PRIMARY KEY AUTO_INCREMENT,
    person_id       BIGINT NOT NULL,
    request_year    INT NOT NULL,
    requested_level INT NOT NULL,
    current_level   INT,
    request_type    ENUM('ผลงาน','บริจาค') NOT NULL,
    work_title      TEXT,
    work_detail     TEXT,
    is_group_work   TINYINT(1) DEFAULT 0,
    donation_amount DECIMAL(15,2),
    donation_purpose VARCHAR(200),
    status          ENUM('draft','submitted','screening','committee_review',
                         'approved','sent_to_cabinet','granted','rejected') DEFAULT 'draft',
    eligibility_passed  TINYINT(1),
    eligibility_notes   TEXT,
    committee_decision  TEXT,
    committee_date      DATE,
    submitted_by    BIGINT,
    submitted_dept  VARCHAR(200),
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (person_id) REFERENCES direk_persons(person_id),
    FOREIGN KEY (requested_level) REFERENCES direk_levels(level_id)
);

-- เอกสารประกอบ
CREATE TABLE direk_documents (
    doc_id          BIGINT PRIMARY KEY AUTO_INCREMENT,
    request_id      BIGINT NOT NULL,
    doc_type        ENUM('นร1','นร2','นร3','นร4','นร5','นร6','อื่นๆ') NOT NULL,
    file_name       VARCHAR(500),
    file_path       VARCHAR(1000),
    file_size       INT,
    uploaded_by     BIGINT,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (request_id) REFERENCES direk_requests(request_id)
);

-- กฎระเบียบที่เกี่ยวข้อง
CREATE TABLE direk_regulations (
    regulation_id   BIGINT PRIMARY KEY AUTO_INCREMENT,
    title           VARCHAR(500) NOT NULL,
    reference_no    VARCHAR(200),
    effective_date  DATE,
    content         TEXT,
    file_path       VARCHAR(1000),
    is_active       TINYINT(1) DEFAULT 1,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```
